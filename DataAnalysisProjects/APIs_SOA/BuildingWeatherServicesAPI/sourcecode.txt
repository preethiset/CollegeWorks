<script src="http://ajax.googleapis.com/ajax/libs/jquery/1.5.1/jquery.min.js"></script>
<script>
jQuery(document).ready(function($) {
  $.ajax({
url : "http://api.wunderground.com/api/2e02969defcb4a64/geolookup/conditions/q/Australia/Sydney.json",

  dataType : "jsonp",
  success : function(parsed_json) {
  var city = parsed_json['location']['city'];
var state = parsed_json['location']['state'];
  var country_name = parsed_json['location']['country_name'];

  var lat = parsed_json['location']['lat'];
  var lon = parsed_json['location']['lon'];
  var temp_f = parsed_json['current_observation']['temp_f'];
  var temp_c = parsed_json['current_observation']['temp_c'];
  var icon_url = parsed_json['current_observation']['icon_url'];
  var weather = document.getElementById('wuDiv');
  weather.innerHTML ="The Co-ordinates of " + city + ","  + country_name + " is: " + "<br/>" + "Latitude:" + lat + " , " + "<br/>" + "Longitude:" + lon + "<br/>" + 
"The state code of "+ city + " is :" +state+ "<br/>" + " Current temperature in " + city + " is: " + temp_f + " F / " + temp_c + " C" + " <img src=" +icon_url+ " />" + "<br/>"   ;
  }
  });
});
</script>
